import java.util.*;
public class Problem7 {

    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int n = sc.nextInt();

        for(int i=1; i<11; i++)
        {
            int res = n * i;
            System.out.println(n+ " * " + i + " = " + res );
        }

    }
}
